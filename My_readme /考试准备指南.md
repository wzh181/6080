### 5.1 App.js åŸºç¡€ç»“æ„

```javascript
import { BrowserRouter, Routes, Route, Link } from 'react-router-dom';
import { useState, useEffect } from 'react';
import './App.css';
import Dashboard from './Dashboard';
import Game1 from './Game1';
import Game2 from './Game2';
import Game3 from './Game3';

function App() {
  const [gamesWon, setGamesWon] = useState(0);

  useEffect(() => {
    const saved = localStorage.getItem('gamesWon');
    if (saved) {
      setGamesWon(parseInt(saved));
    } else {
      // ä»APIè·å–åˆå§‹å€¼
      fetch('https://cgi.cse.unsw.edu.au/~cs6080/raw/data/score.json')
        .then(res => res.json())
        .then(data => {
          setGamesWon(data.score);
          localStorage.setItem('gamesWon', data.score);
        });
    }
  }, []);

  return (
    <BrowserRouter>
      <Header />
      <Routes>
        <Route path="/" element={<Dashboard gamesWon={gamesWon} setGamesWon={setGamesWon} />} />
        <Route path="/game1" element={<Game1 setGamesWon={setGamesWon} />} />
        <Route path="/game2" element={<Game2 setGamesWon={setGamesWon} />} />
        <Route path="/game3" element={<Game3 setGamesWon={setGamesWon} />} />
      </Routes>
      <Footer />
    </BrowserRouter>
  );
}

export default App;
```
